CSS Cheatsheet

/* Reset Default Page Styles */
* {
    margin: 0;
    padding: 0;
}

/* Importing Styles */
@import "something.css";
/* Importing an external stylesheet */

/* Text and Fonts */
font: italic bold 36px arial;
/* Sets font properties: style, weight, size, and family */

/* Borders and Boxes */
border: 4px solid red;
/* Sets border properties: width, style, and color */

/* Backgrounds */
background: ;
/* Sets background properties: color, image, position, size, repeat, origin, clip, attachment */
list-style: square inside url('a.png');
/* Sets list properties: style, position, and image */
list-style-imgage: url('a.png');
/* Sets the image for list item markers */

/* Alignment and Positioning */
vertical-align: middle, baseline, top, 5px, ...;
/* Vertical alignment options */

/* Colors and Styling Links */
background-color;
/* Sets the background color */
background-image: url('images/grass.png');
/* Sets the background image */
background-position: 40px 70px;
/* Sets the background position */
background-size: cover, ...px, contain;
/* Sets the background size */
background-repeat: no-repeat;
/* Sets background repeat */
background-origin: border-box, ...;
/* Specifies where the background image's position is calculated */
background-clip: border-box, content-box, ...;
/* Specifies how far the background extends within an element */
background-attachment: scroll, local, fixed;
/* Sets whether the background image is fixed or scrolls with the rest of the page */

/* Tables */
border-spacing: ...px, 0;
/* Sets the distance between cells in a table */
border-collapse: collapse;
/* Merges borders into a single border for adjacent cells in a table */
text-align: center;
/* Aligns text within an element */
caption-side: bottom, top, ...;
/* Sets the placement of a table caption */
cursor: default;
/* Sets the type of cursor to be displayed when hovering over an element */

input[type="radio, text, password, submit, reset,..."] {
    /* Various input types with the 'required' attribute */
}
:checked {
    /* Styling for checked radio buttons */
}

/* Links and Pseudo-classes */
a:hover {
    color: red;
    text-decoration: none;
}
/* Hover effect for links */
a:link {
    background-color: gold;
    color: green;
}
/* Styles for unvisited links */
a:visited {
    color: red;
    background-color: gray;
}
/* Styles for visited links */
/* Styles for  tag with specific attributes */
a[title] {
    color: blue;
}
a[href="https://google.com"] {
    color: magenta;
}
a[href*="google"] {
    font-style: italic;
}

/* Miscellaneous Selectors */
ul.fruits li a {
    text-decoration: none;
}
/* Styles for links inside a specific list */
h1 + p {
    color: green;
    font-size: 20px;
    text-decoration: underline;
}
/* Styles for the first 
 element that directly follows an 
 */
ul > li {
    list-style: disc;
}
/* Styles for list items directly under a 
     */
li:nth-child(2) {
    color: blue;
}
/* Styles for the second 
  •  element in a list */
:nth-child(4n) {
    text-decoration: underline;
}
/* Styles for every fourth element among any group of siblings */
ul:nth-child(even) {
    color: gray;
}
/* Styles for every even element among any group of siblings */
div:nth-of-type(3) p {
    background-color: pink;
}
